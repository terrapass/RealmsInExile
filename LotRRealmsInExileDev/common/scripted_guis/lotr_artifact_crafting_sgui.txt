### Artifact related sGUI's ###
###############################

can_craft_artifacts = {
	is_shown = {
		always = yes
	}
	
	is_valid = {
		OR = {
			has_trait = lifestyle_artifact_crafting
		}
	}
}

select_no_artifact_type = {
	is_valid = {
		has_variable = artifact_craft
	}
	effect = {
		debug_log = " Artifact Var removed "
		remove_variable = artifact_craft
		remove_variable = used_material
	}
}

select_weapon_artifact_type = {
	effect = {
		debug_log = " Weapon Var added "
		set_variable = {
			name = artifact_craft
			value = flag:weapon_artifact
		}
		set_variable = {
			name = used_material
			value = flag:no_material
		}
	}
}

display_material_available = {
	is_valid = {
		OR = {
			has_variable = material_mithril
			has_variable = material_gold
			has_variable = material_silver
			has_variable = material_tin
			has_variable = material_lead
			has_variable = material_steel
			# has_variable = material_iron
			has_variable = material_bronze
			has_variable = material_copper
		}
	}
}

mithril_crafting_county = {
	is_valid = {
		OR = {
			any_held_title = {
				title_province = { has_province_modifier = prospecting_mithril_province_modifier }
			}
			any_vassal_or_below = {
				any_held_title = {
					title_province = { has_province_modifier = prospecting_mithril_province_modifier }
				}
			}
		}
	}
}
