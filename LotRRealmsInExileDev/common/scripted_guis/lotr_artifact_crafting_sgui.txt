### Local Scripted Effects ###
##############################

scripted_effect add_artifact_crafting_material_var = {
	add_to_variable_list = {
		name = artifact_crafting_modifiers
		target = var:used_material
	}
}

scripted_effect remove_artifact_crafting_material_var = {
	remove_list_variable = {
		name = artifact_crafting_modifiers
		target = var:used_material
	}
}

can_craft_artifacts = {
	is_shown = {
		always = yes
	}
	
	is_valid = {
		OR = {
			has_trait = lifestyle_artifact_crafting
		}
	}
}

select_no_artifact_type = {
	is_valid = {
		has_variable = artifact_craft
	}
	effect = {
		debug_log = " Artifact Var removed "
		remove_variable = artifact_craft
		remove_variable = used_material
		clear_variable_list = artifact_crafting_modifiers
	}
}

select_weapon_artifact_type = {
	effect = {
		debug_log = " Weapon Var added "
		set_variable = {
			name = artifact_craft
			value = flag:weapon_artifact
		}
		set_variable = {
			name = used_material
			value = flag:no_material
		}
	}
}
