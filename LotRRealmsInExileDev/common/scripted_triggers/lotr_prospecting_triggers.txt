
can_mine_trigger = {
	NOR = {
		has_trait = wounded_2
		has_trait = wounded_3
		has_trait = infirm
		has_trait = maimed
		has_trait = blind
	}
	has_contagious_deadly_disease_trigger = no
}

miner_lifestyle_track_less_trigger = {
	has_trait = lifestyle_miner
	has_trait_xp = {
		trait = lifestyle_miner
		track = $TRACK$
		value < $LESS_THAN$
	}
}

miner_lifestyle_track_greater_equal_trigger = {
	has_trait = lifestyle_miner
	has_trait_xp = {
		trait = lifestyle_miner
		track = $TRACK$
		value >= $GREATER_EQUAL$
	}
}
