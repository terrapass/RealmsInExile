#Triggers for races of LOTR
#these should work for character, culture and province scopes unless stated otherwise

is_human = { #all other races and species should be added here
	is_orc = no
	is_elf = no
}

is_orc = {
	culture_group = culture_group:lotr_melkor_creatures_group
}

is_elf = {
	OR = {
		culture_group = culture_group:lotr_amanyar_group
		culture_group = culture_group:lotr_umanyar_group
		culture_group = culture_group:lotr_quendi_group
	}
}

same_race = { #compares two scopes, example usage same_race = { TARGET = scope:target_ruler }
	OR = {
		AND = {
			is_human = yes
			$TARGET$ = { is_human = yes }
		}
		AND = {
			is_orc = yes
			$TARGET$ = { is_orc = yes }
		}
		AND = {
			is_elf = yes
			$TARGET$ = { is_elf = yes }
		}
	}
}

culture_conversion_from_allowed = { #only checks if the culture can convert others at all, for comparing two cultures use the next one
	NOT = { culture_group = culture_group:lotr_maia_group }
}

culture_conversion_to_allowed = { #example usage culture_conversion_to_allowed = { TARGET = scope:recipient.culture }
	culture_conversion_from_allowed = yes
	same_race = { TARGET = $TARGET$ }
}
