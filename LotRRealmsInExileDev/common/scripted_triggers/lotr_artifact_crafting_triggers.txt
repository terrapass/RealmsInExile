contains_prospecting_modifier = {
	OR = {
		any_held_title = {
			title_province ?= { has_province_modifier = $PROSPECTING_MODIFIER$ }
		}
		any_vassal_or_below = {
			any_held_title = {
				title_province ?= { has_province_modifier = $PROSPECTING_MODIFIER$ }
			}
		}
	}
}

valid_increase_base_stat_modifier = {
	OR = {
		trigger_if = {
			limit = { exists = var:$BASE_STAT$ }
			var:$BASE_STAT$ < 5
		}
		NOT = {
			exists = var:$BASE_STAT$
		}
	}
}

valid_decrease_base_stat_modifier = {
	OR = {
		trigger_if = {
			limit = { exists = var:$BASE_STAT$ }
			var:$BASE_STAT$ > 0
		}
	}
}

valid_increase_enchanting_modifier = {
	OR = {
		trigger_if = {
			limit = { exists = var:$MODIFIER$ }
			var:$MODIFIER$ < 5
		}
		NOT = {
			exists = var:$MODIFIER$
		}
	}
}

valid_dencrease_enchanting_modifier = {
	OR = {
		trigger_if = {
			limit = { exists = var:$MODIFIER$ }
			always = yes
		}
	}
}
