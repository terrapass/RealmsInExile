##################################################
# Tulwang Struggle Triggers
 
character_any_involvement_tulwang_struggle_trigger = {
	has_fp2_dlc_trigger = yes
	OR = {
		character_involved_in_tulwang_struggle_trigger = yes
		character_interloper_in_tulwang_struggle_trigger = yes
	}
}

character_involved_in_tulwang_struggle_trigger = {
	has_fp2_dlc_trigger = yes
	any_character_struggle = {
		involvement = involved
		is_struggle_type = tulwang_struggle
	}
}

character_interloper_in_tulwang_struggle_trigger = {
	has_fp2_dlc_trigger = yes
	any_character_struggle = {
		involvement = interloper
		is_struggle_type = tulwang_struggle
	}
}

character_uninvolved_in_tulwang_struggle_trigger = {
	has_fp2_dlc_trigger = yes
	exists = struggle:tulwang_struggle
	character_involved_in_tulwang_struggle_trigger = no
	character_interloper_in_tulwang_struggle_trigger = no
}

tulwang_struggle_reclamation_involved_county_trigger = {
	any_realm_county = {
		any_county_struggle = { is_struggle_type = tulwang_struggle }
	}
}

tulwang_struggle_reclamation_should_vassalize_trigger = {
	any_character_struggle = { is_struggle_type = tulwang_struggle }
	any_held_title = { tier < scope:attacker.primary_title.tier }
	any_realm_county = {
		count = all
		duchy = { is_in_list = target_titles }
		any_county_struggle = { is_struggle_type = tulwang_struggle }
	}
}

tulwang_reclamation_vassalize_vassal_trigger = {
	primary_title.tier >= tier_county
	primary_title.tier < scope:attacker.primary_title.tier
	any_character_struggle = { is_struggle_type = tulwang_struggle }
	capital_county = {
		duchy = { is_in_list = target_titles }
	}
	any_sub_realm_county = {
		count = all
		duchy = { is_in_list = target_titles }
		any_county_struggle = { is_struggle_type = tulwang_struggle }
	}
}
