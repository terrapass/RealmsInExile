# Makes wasteland rulers gods (trait wastelands)
remove_traits_wastelands_effect = {
	every_living_character = {
		limit = {
			has_culture = culture:wastelands
		}
        clear_traits = yes
        add_trait = wastelands
		add_character_flag = is_naked
	}		
}

# Destroy at game start everyone in the wastelands areas
empty_wastelands_at_start_effect = {
	every_living_character = {
		limit = {
			has_culture = culture:wastelands
		}
		death = {
			death_reason = death_vanished
		}
	}		
}

empty_wastelands_effect = {
	every_living_character = {
		limit = {
			is_ruler = no
			OR = {
				has_culture = culture:wastelands
				has_government = no_government
			}			
		}
		death = {
			death_reason = death_vanished
		}
	}		
}


empty_wastelands_ruler_effect = {
	every_living_character = {
		limit = {
			is_ruler = yes
			OR = {
				has_culture = culture:wastelands
				has_government = no_government
			}			
		}
		death = {
			death_reason = death_vanished
		}
	}		
}


empty_people_in_wastelands_effect = {
	every_living_character = {
		limit = {
			is_ruler = no
			exists = location
			location.culture = culture:wastelands
		}
		death = {
			death_reason = death_vanished
		}
	}		
}
