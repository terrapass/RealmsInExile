		target = var:used_material
	}
}

	remove_list_variable ?= {
		name = artifact_crafting_modifiers
		target = var:used_material
	}
}

### Base Stat Effects ####
add_artifact_crafting_base_stat_modifier = {
	if = {
		limit = {
			$VALUE$ = 0
		}
		remove_list_variable = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_1
		}
	}
	else_if = {
		limit = {
			$VALUE$ = 1
		}
		remove_list_variable = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_2
		}
		add_to_variable_list = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_1
		}
	}
	else_if = {
		limit = {
			$VALUE$ = 2
		}
		remove_list_variable = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_3
		}
		remove_list_variable = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_1
		}
		add_to_variable_list = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_2
		}
	}
	else_if = {
		limit = {
			$VALUE$ = 3
		}
		remove_list_variable = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_4
		}
		remove_list_variable = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_2
		}
		add_to_variable_list = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_3
		}
	}
	else_if = {
		limit = {
			$VALUE$ = 4
		}
		remove_list_variable = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_5
		}
		remove_list_variable = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_3
		}
		add_to_variable_list = {
			name = artifact_crafting_modifiers
			target = flag:$NAME$_4
		}
	}
	else = {
		if = {
			limit = {
				$VALUE$ = 5
			}
			remove_list_variable = {
				name = artifact_crafting_modifiers
				target = flag:$NAME$_4
			}
			add_to_variable_list = {
				name = artifact_crafting_modifiers
				target = flag:$NAME$_5
			}
		}
	}
}

artifact_crafting_modifier_effect_decrease = {
	decrease_variable = {
		NAME = $VALUE$
		AMOUNT = 1
	}
	save_scope_value_as = {
		name = $VALUE$ue
		value = var:$VALUE$
	}
	add_artifact_crafting_base_stat_modifier = {
		NAME = $NAME$
		VALUE = scope:$VALUE$ue
	}
}

artifact_crafting_modifier_effect_increase = {
	increase_variable = {
		NAME = $VALUE$
		AMOUNT = 1
	}
	save_scope_value_as = {
		name = $VALUE$ue
		value = var:$VALUE$
	}
	add_artifact_crafting_base_stat_modifier = {
		NAME = $NAME$
		VALUE = scope:$VALUE$ue
	}
}
