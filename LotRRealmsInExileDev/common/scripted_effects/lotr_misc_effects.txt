increase_variable = { # From Tobbzn
    if = {limit = {NOT = {has_variable = $NAME$}}
        set_variable = {name = $NAME$ value = 0}
    }
    change_variable = {
        name = $NAME$
        add = $AMOUNT$
    }
}

decrease_variable = { # From Tobbzn
    if = {limit = {NOT = {has_variable = $NAME$}}
        set_variable = {name = $NAME$ value = 0}
    }
    change_variable = {
        name = $NAME$
        subtract = $AMOUNT$
    }
}
