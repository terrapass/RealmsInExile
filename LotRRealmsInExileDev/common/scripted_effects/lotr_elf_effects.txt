### ELF EFFECTS ###
change_elf_government_effect = {
	if = {
		limit = { is_elf = yes }
		if = {
			limit = { NOT = { capital_province = { has_holding = elven_holding } } }
			capital_province = { set_holding_type = elven_holding }
			add_prestige = medium_prestige_loss
			if = {
				limit = { NOT = { has_government = elven_government } }
				change_government = elven_government
			}
		}
	}
}
