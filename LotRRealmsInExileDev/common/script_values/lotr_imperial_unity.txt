iu_annual_change = {
	value = 0
	add = iu_vassals_that_like_me
	subtract = iu_vassals_that_dislike_me
}

iu_vassals_that_like_me = {
	every_vassal = {
		limit = { 
			opinion = {
				target = prev
				value > 0
			}
		}
		add = 1
	}
}

iu_vassals_that_dislike_me = {
	every_vassal = {
		limit = { 
			opinion = {
				target = prev
				value < 0
			}
		}
		add = 1
	}
}
