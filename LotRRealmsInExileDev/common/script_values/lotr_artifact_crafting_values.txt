calc_success_chance = {
	value = 0
	if = { # So that it displays zero percent chance, instead of going into negatives
		limit = { calc_zero_percent_chance < 0 }
		add = 0
	}
	else = {
		add = calc_zero_percent_chance
	}
}

calc_zero_percent_chance = {
	value = 0

	add = calc_out_of_100_success_chance
	add = calc_out_of_100_crit_success_chance
	# subtract = calc_out_of_100_failure_chance
	# subtract = calc_out_of_100_crit_failure_chance
}

calc_out_of_100_success_chance = {
	value = 0

	add = {
		value = calc_edited_success_chance
		divide = total_success_failure_chances
		multiply = 100
	}
}
calc_out_of_100_crit_success_chance = {
	value = 0
	
	add = {
		value = calc_edited_critical_success_chance
		divide = total_success_failure_chances
		multiply = 100
	}
}
calc_out_of_100_failure_chance = {
	value = 0
	
	add = {

}
calc_out_of_100_crit_failure_chance = {
	value = 0

	add = {
		value = calc_edited_critical_failure_chance

total_success_failure_chances = {
	value = 0

	add = calc_edited_success_chance

}

	}
}
