### NÚMENÓREAN BLOOD VALUES ###
get_numenorean_blood_tier = {
	value = 0
	if = {
		limit = {
			has_trait_rank = {
				trait = blood_of_numenor
				rank = 1
			}
		}
		add = 1
	}
	else_if = {
		limit = {
			has_trait_rank = {
				trait = blood_of_numenor
				rank = 2
			}
		}
		add = 2
	}
	else_if = {
		limit = {
			has_trait_rank = {
				trait = blood_of_numenor
				rank = 3
			}
		}
		add = 3
	}
	else_if = {
		limit = {
			has_trait_rank = {
				trait = blood_of_numenor
				rank = 4
			}
		}
		add = 4
	}
}

numenorean_blood_tier = { value = get_numenorean_blood_tier }
