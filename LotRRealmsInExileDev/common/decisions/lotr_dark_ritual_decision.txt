learn_sorcery_decision = {
	title = learn_sorcery_decision
	picture = "gfx/interface/illustrations/decisions/decision_magic_02.dds"

	major = no
	ai_goal = yes

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 50
	}

	desc = learn_sorcery_decision_desc
	selection_tooltip = learn_sorcery_decision_tooltip
	confirm_text = learn_sorcery_decision_confirm

	is_shown = {
		is_landed = yes
		religion = religion:forces_of_evil_religion
		NOR = {
			is_imprisoned = yes
			has_character_flag = planning_an_activity
			has_character_flag = learning_sorcery
			has_character_modifier = studying_dark_arts_modifier
			has_character_modifier = sorcery_connoisseur_modifier
			has_trait = sorcerer
			has_trait = nazgul
			has_trait = istari
			has_trait = maiar
		}
		OR = {
			is_ai = no
			AND = {
				is_ai = yes
				calc_true_if = {
    				amount >= 2
					has_trait = ambitious
					has_trait = zealous
					has_trait = education_learning_4
					learning >= 24
				}
			}
		}
	}

	is_valid = {
		piety_level >= 2
		NOT = {
			has_trait = sorcerer_1
		}
		OR = {
			has_trait = ambitious
			has_trait = zealous
		}
		learning >= 4
	}

	effect = {
		custom_tooltip = learn_sorcery_decision_desc_activity
		custom_tooltip = learn_sorcery_decision_desc_benefits
		trigger_event = {
			id = evil.0100
		}
	}
}
