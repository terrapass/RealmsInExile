decision_challenge_the_orc_leader = {

	title = decision_challenge_the_orc_leader
	picture = "gfx/interface/illustrations/decisions/decision_orcs_challenge.dds"
	major = no

	ai_check_interval = 1
	ai_goal = yes
	ai_will_do = {
		base = 0
		opinion_modifier = {
			opinion_target = root.liege
			who = root
			multiplier = -1
		}
		modifier = {
        	add = { 
        		value = root.prowess
        		subtract = root.liege.prowess
        		multiply = 10 
        	}
        }
		modifier = {	
			add = 50
			AND = {
				has_trait = ambitious
				root.prowess > root.liege.prowess
			}
		}
		modifier = {
			add = -30
			root.prowess < root.liege.prowess
		}
		modifier = {
			add = 30
			has_trait = wrathful
		}
		modifier = {
			add = -100
			NOR = {
				has_trait = content
				has_trait = craven
			}
		}
	}

	selection_tooltip = decision_challenge_the_orc_leader_tooltip
	desc = decision_challenge_the_orc_leader_desc
	confirm_text = decision_challenge_the_orc_leader_confirm

	cooldown = { years = 5 }

	is_shown = {
		root = {
			is_orc = yes
		}
	}

	is_valid = {
		prestige_level >= low_prestige_level

		root.liege = {
			is_adult = yes
			is_orc = yes
			NOR = {has_character_modifier = orc_challenge_win_modifier
			has_character_modifier = orc_challenge_rejected_modifier}
			is_imprisoned = no
		}
		root = {
			NOT = {has_character_modifier = orc_challenge_retired_modifier}
			is_imprisoned = no
		}
	}

	cost = {
		prestige = {
			value = 500 #major_prestige_value
			if = {
				limit = {
					is_ai = yes
				}
				multiply = 0
			}
		}
	}

	effect = {
		custom_tooltip = decision_challenge_the_orc_leader_tooltip
		trigger_event = {
			id = orc_challenge.0001
		}
	}
}
