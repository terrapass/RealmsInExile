decision_form_dunland_kingdom_vassal = {
	title = decision_form_dunland_kingdom_vassal
	picture = "gfx/interface/illustrations/decisions/decision_wildmen_03.dds"
	major = yes
	ai_goal = yes
	selection_tooltip = decision_form_dunland_kingdom_vassal_tooltip
	desc = decision_form_dunland_kingdom_vassal_desc
	confirm_text = decision_form_dunland_kingdom_vassal_confirm

	is_shown = {
		has_character_flag = saruman_dunlending_auxilliaries
		title:k_dunland = {
			is_title_created = no
		}
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:decision_form_dunland_kingdom_vassal
			}
		}
		any_vassal = { has_character_flag = promised_kingdom_of_dunland }
	}

	is_valid = {
		custom_description = {
			text = k_dunland_is_creatable
			any_alert_creatable_title = {
				this = title:k_dunland
			}
		}
	}

	effect = {
		if = {
			limit = {
				title:e_isengard_unleashed = { is_title_created = no }
				highest_held_title_tier < tier_empire
			}
			create_title_and_vassal_change = {
				type = created
				save_scope_as = title_change
				add_claim_on_loss = no
			}
			title:e_isengard_unleashed = {
				change_title_holder = {
					holder = root
					change = scope:title_change
				}
			}
			resolve_title_and_vassal_change = scope:title_change
		}
		random_vassal = {
			limit = { has_character_flag = promised_kingdom_of_dunland }
			save_scope_as = recipient
			create_title_and_vassal_change = {
				type = created
				save_scope_as = title_change
				add_claim_on_loss = no
			}
			title:k_dunland = {
				change_title_holder = {
					holder = scope:recipient
					change = scope:title_change
				}
			}
			resolve_title_and_vassal_change = scope:title_change
		}
		every_held_title = {
			limit = {
				tier = tier_county
				kingdom = title:k_dunland
			}
			create_title_and_vassal_change = {
				type = granted
				save_scope_as = title_change
				add_claim_on_loss = no
			}
			change_title_holder = {
				holder = scope:recipient
				change = scope:title_change
			}
			resolve_title_and_vassal_change = scope:title_change
		}
		add_prestige = 500
		scope:recipient = { add_character_modifier = unifier_of_the_dunlendings_character_modifier }
		add_character_modifier = friend_of_the_dunlendings_character_modifier
		capital_county = { add_county_modifier = loyal_dunlendings_of_isengard_county_modifier }
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:decision_form_dunland_kingdom_vassal
		}
	}
	
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 100
	}
}
