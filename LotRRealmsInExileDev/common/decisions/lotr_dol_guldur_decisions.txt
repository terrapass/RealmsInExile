decision_swaying_evil_woodmen = {
	title = decision_swaying_evil_woodmen
	desc = decision_swaying_evil_woodmen_desc
	confirm_text = decision_swaying_evil_woodmen_confirm
	selection_tooltip = decision_swaying_evil_woodmen_tooltip
	picture = "gfx/interface/illustrations/decisions/decision_woodmen.dds"
	major = yes

	ai_check_interval = 60

	# Shown
	is_shown = {
		is_landed = yes
		has_title = title:k_amon_lanc
		has_faith = faith:lidless_eye

		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:decision_swaying_evil_woodmen
			}
		}
	}

	# Effect
	effect = {
		custom_tooltip = decision_swaying_evil_woodmen.send_message
		every_independent_ruler = {
			limit = {
				NOT = {
					this = ROOT
				}
				primary_title = { kingdom = title:k_amon_lanc }
				faith_is_evil = yes
				culture != culture:wastelands
			}
			trigger_event = {
				id = dol_guldur.0001
				days = { 5 10 }
			}
		}
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:decision_creation_of_the_uruk_hai
		}
	}

	# Requirements
	is_valid = {
		is_ruler = yes
		has_faith = faith:lidless_eye
		has_title = title:k_amon_lanc
		NOT = {
			completely_controls = title:k_amon_lanc
		}

		custom_tooltip = {
			text = decision_swaying_evil_woodmen.check_evil_trigger
			any_independent_ruler = {
				NOT = {
					this = ROOT
				}
				primary_title = { kingdom = title:k_amon_lanc }
				faith_is_evil = yes
				culture != culture:wastelands
			}
		}

	}

	# Cost
	cost = {
		gold = 50
		prestige = 100
	}

	# AI
	ai_check_interval = 12

	ai_potential = {

	}

	ai_will_do = {
		base = 50
	}
}
