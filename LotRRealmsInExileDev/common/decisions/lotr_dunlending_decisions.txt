
decision_create_dunland_kingdom = {
	title = decision_create_dunland_kingdom
	picture = "gfx/interface/illustrations/decisions/decision_dunland_01.dds"
	major = yes
	ai_goal = no
	ai_check_interval = 120
	selection_tooltip = decision_create_dunland_kingdom_tooltip
	desc = decision_create_dunland_kingdom_desc
	confirm_text = decision_create_dunland_kingdom_confirm
	
	is_shown = {
		OR = { # Only the Dunlendings should care about restoring the old Dunland kingdom
			has_culture = culture:herdfolk_dunlending 
			has_culture = culture:hill_dunlending 
		}
		is_landed = yes
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:decision_create_dunland_kingdom
			}
		}
	}
	is_valid = {
		completely_controls = title:d_dunlandfells
		completely_controls = title:d_dunfearn
		completely_controls = title:d_dathabha
		completely_controls = title:d_dunlending_wilds
		completely_controls = title:d_westmarches
		completely_controls = title:d_westfold
		completely_controls = title:d_thefolde
	}
	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
		is_at_war = no
	}
	
	effect = {
		add_unpressed_claim = d_isengard
		add_unpressed_claim = k_rohan
	
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:decision_create_dunland_kingdom
		}

		show_as_tooltip = {
			decision_create_dunland_kingdom_effect = yes
		}
		trigger_event = dunland.1001
		save_scope_as = restorer
		# Send event to other players
		every_player = {
			limit = {
				NOT = { this = root }
				OR = {
					primary_title = {
						title_capital_county.title_province = {  geographical_region = special_rohirrim_t4a_kingdom }
					}
					has_culture_group = culture_group:lotr_dunlending_group
#					has_culture_group = culture_group:lotr_andrast_gwathuirim_group
#					has_culture_group = culture_group:lotr_druwaith_iaur_gwathuirim_group
#					has_culture_group = culture_group:lotr_arnorian_gwathuirim_group
					has_culture = culture:eotheod
					has_culture = culture:rohirrim
#					has_culture = culture:gondorian
				}
			}
			trigger_event = dunland.1002
		}
	}
	
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 100
	}
}

decision_restore_saralain = { # If Saralain has been destroyed
	title = decision_restore_saralain
	picture = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds"
	major = yes
	ai_goal = no
	ai_check_interval = 120
	selection_tooltip = decision_restore_saralain_tooltip
	desc = decision_restore_saralain_desc
	confirm_text = decision_restore_saralain_confirm
	
	is_shown = { # Can be seen by independent Saralain lords if the Kingdom of Saralain has no holder
		has_culture = culture:saralain
		is_landed = yes
		title:k_saralain = { exists = no } # If Saralain is not held by anyone...
	}
	is_valid = { 
		highest_held_title_tier >= duchy # Only a duke or above
		root = {
			prestige_level >= 4 # Very prestigous and well respected
		}
	}
	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
		is_at_war = no
	}
	
	effect = {
		get_title = title:k_saralain
		root = {
			save_scope_as = saralain_claimer
		}
		every_living_character = { # Other independent Saralain dukes gets a chance to swear fealty or try to claim the title themselves
			limit = {
				is_independent_ruler = yes
				has_culture = culture:saralain
				highest_held_title_tier = tier_duchy
			}
			trigger_event = {
				id = enedwaith.0200
				days = { 30 90 }
			}
		}
		every_living_character = { # Other independent Saralain counts gets a chance to swear fealty or remain independent
			limit = {
				is_independent_ruler = yes
				has_culture = culture:saralain
				highest_held_title_tier = tier_county
			}
			trigger_event = {
				id = enedwaith.0201
				days = { 30 90 }
			}
		}
		custom_tooltip = restore_saralain_custom_warning
		
	}
	
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 100
	}
}
