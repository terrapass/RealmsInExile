
colonize_land = {
	category = interaction_category_colonization	
	desc = colonize_land_desc
	auto_accept = yes
	use_diplomatic_range = yes
	ignores_pending_interaction_block = yes

	is_shown = {
		scope:actor = {			
			lotr_wastelands_trigger = no
			
			# Only surrounding lands
			any_sub_realm_county = {
				is_neighbor_to_realm = scope:recipient
			}
		}
		
		scope:recipient = {
			lotr_wastelands_trigger = yes
		}
	}

	is_valid_showing_failures_only = {	
		scope:actor = {		
			# I have money and prestige
			gold >= 100
			prestige >= 100
			
			# Only top liege
			is_independent_ruler = yes
		}
	}

	on_accept = {
		
		scope:recipient.primary_title = {
			save_scope_as = recipient_title
		}
		scope:recipient.primary_title.county = {
			save_scope_as = recipient_county
		}
		scope:recipient.location = {
			save_scope_as = recipient_location
		}
		
		scope:actor = {
			send_interface_toast = {
				title = colonization_started_title
				left_icon = root		
				
				remove_short_term_gold = 100
				add_prestige = -100
				
				add_character_modifier = {
					modifier = colonization_cooldown
					days = 1325 # 5 years
				}
				
				add_character_flag = {
					flag = colonization_cooldown_flag
					years = 5
				}
			}
			
		}				
		create_title_and_vassal_change = {
  			type = granted
  			save_scope_as = change
  		}
  		scope:recipient_title = {
	  		change_title_holder = {
				holder = scope:actor
				change = scope:change
				take_baronies = yes
			}
			set_county_faith = scope:actor.faith
			set_county_culture = scope:actor.culture
  		}
		resolve_title_and_vassal_change = scope:change
		scope:recipient_location = {			
			set_holding_type = settlement_holding
		}
	}
}
