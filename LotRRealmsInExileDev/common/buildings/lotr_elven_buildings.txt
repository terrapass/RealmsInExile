	construction_time = standard_construction_time

	can_construct_potential = {
		scope:holder = { is_elf = yes }
		has_building_or_higher = elven_settlement_01
		lotr_elven_forest_requirement_terrain = yes
		OR = { # Flets are a tradition of Lorien and Greenwood
			galadhrim_trigger = yes 
			greenwood_trigger = yes
		}
	}

	can_construct_showing_failures_only = {
		building_requirement_tribal = no
	}
	
	cost_gold = cheap_building_tier_1_cost

		garrison_size = 0.05
		max_garrison = 100
	}
	
	character_modifier = {
		archers_damage_mult = 0.05

		archers_max_size_add = 1
	}
	
	ai_value = {

		ai_economical_building_preference_modifier = yes
	}
	
	type_icon = "icon_building_hunting_grounds.dds"
	
	next_building = tree_flet_02
}
