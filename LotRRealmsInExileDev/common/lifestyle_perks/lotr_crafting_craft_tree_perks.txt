# craft_1_perk --> general increase in the XP gained when an artifact has been crafted
craft_1_perk = {
	lifestyle = crafting_lifestyle
	tree = craft
	position = { 2 0 }
	icon = node_crafting
	
	auto_selection_weight = {
		value = 11
	}
	
	effect = {
		custom_description_no_bullet = {
			text = increased_crafting_xp
		}
		if = {
			limit = { NOT = { has_trait = lifestyle_artifact_crafting } }  
			add_trait = lifestyle_artifact_crafting
		}
	}
}
