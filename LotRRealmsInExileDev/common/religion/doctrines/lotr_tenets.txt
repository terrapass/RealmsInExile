doctrine_core_tenets = {
	group = "core_tenets"
	number_of_picks = 3

	tenet_unforgetting = { # Dunlending warmonger tenet
		icon = core_tenet_sword
		piety_cost = {
			if = {
				limit = {
					religion_tag = free_peoples_religion
				}
				value = faith_tenet_cost_massive
			}
			else = {
				value = faith_tenet_cost_mid
			}
			if = {
				limit = { has_doctrine = tenet_unforgetting }
				multiply = faith_unchanged_doctrine_cost_mult
			}
		}
		is_shown = {
			religion_tag = free_peoples_religion
		}
		can_pick = {
			custom_description = {
				text = incompatible_tenet_pacifism_trigger
				NOT = { flag:tenet_pacifism = { is_in_list = selected_doctrines } }
			}
			custom_description = {
				text = incompatible_tenet_dharmic_pacifism_trigger
				NOT = { flag:tenet_dharmic_pacifism = { is_in_list = selected_doctrines } }
			}
		}

		character_modifier = {
			diplomacy = -2
			prowess = 2
			ai_war_chance = 1
			ai_war_cooldown = -1
		}
		traits = {
			virtues = { forgiving = 2 }
			sins = { vengeful reaver }
		}
		parameters = {
			warmonger = yes
			conquest_cb_enabled = yes
			invasion_cb_enabled = yes
			great_holy_wars_active_if_reformed = yes
			clergy_can_fight = yes
			piety_from_long_peace = -0.01
		}
	}

	tenet_unquiet_waters = { # Dunlending reincarnation tenet
		icon = core_tenet_adorcism
		piety_cost = {
			if = {
				limit = {
					religion_tag = free_peoples_religion
				}
				value = faith_tenet_cost_high
			}
			else_if = {
				limit = {
					religion_tag = darkness_religion
				}
				value = faith_tenet_cost_low
			}
			else = {
				value = faith_tenet_cost_mid
			}
			if = {
				limit = { has_doctrine = tenet_unquiet_waters }
				multiply = faith_unchanged_doctrine_cost_mult
			}
		}
		is_shown = {
			religion_tag = free_peoples_religion
		}

		character_modifier = {
			# Increase stress and dread gain if holding is coastal.
			# scope:county is_coastal = yes
			stress_gain_mult = 0.1
			dread_gain_mult = 0.1
		}
		traits = {
			virtues = { reincarnation = 2 }
			sins = { adventurer = 2 }
		}
		parameters = {
			reincarnation_resistance_to_conversion = yes
			reincarnation_events_active = yes
		}
	}
}
