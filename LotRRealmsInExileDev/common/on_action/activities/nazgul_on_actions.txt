# On actions for the Nazgûl personality events
on_artifact_changed_owner = {
	if = {
		limit = {
			has_variable = ring_of_men
			scope:owner = { is_nazgul = no }
		}
		scope:owner = { #The Nazgûl starts his manhunt
			trigger_event = {
				id = nazgul.0001
				days = { 180 225 }
			}
		}
		if = {
			limit = { 
				scope:old_owner = {
					is_alive = yes
					is_nazgul = no
				}
			}
			scope:old_owner = { #The Precious is lost !
				trigger_event = {
					id = nazgul.0300
					days = 5
				}
			}
		}
	}
}
