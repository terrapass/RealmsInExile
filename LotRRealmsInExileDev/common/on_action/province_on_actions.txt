# root: The province in which the building was completed
on_building_completed = {
	trigger = {
		has_building = obstacle_03
	}
	
	effect = {
		if = {
			limit = {
				county.holder = {
					has_government = feudal_government
				}				
			}
			set_holding_type = castle_holding
		}
		else = {						
			set_holding_type = tribal_holding
		}		
	}
}
