namespace = no_government

# DOWNGRADING (REMOVING TRAITS) TO NEWLY GENERATED LEADERS
no_government.0002 = {
    type = character_event
 	hidden = yes
 	immediate = {        
        clear_traits = yes
        add_trait = wastelands
		add_character_flag = is_naked
    }
 }

# root, scope:guest: the guest
# host: their host (owner of the court)
# characters: list of all characters moving together. Includes the root character.
 no_government.0003 = {
	hidden = yes
	
	trigger = {	
		scope:host = {
			OR = {
				has_culture = culture:wastelands
				has_government = no_government
			}			
		}
	}

	immediate = {
		#scope:host = {
		#	death = {
		#		death_reason = death_vanished
		#	}
		#}
		scope:guest = {
			death = {
				death_reason = death_vanished
			}
		}
		every_in_list = {
			list = characters
			death = {
				death_reason = death_vanished
			}
		}		
		empty_wastelands_effect = yes
		# empty_wastelands_ruler_effect = yes
		empty_people_in_wastelands_effect = yes
		remove_traits_wastelands_effect = yes
		trigger_event = {
			id = no_government.0002
			days = 5
		}
	}
}
