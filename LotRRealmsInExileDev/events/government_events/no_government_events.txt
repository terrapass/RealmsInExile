namespace = no_government

# DOWNGRADING (REMOVING TRAITS) TO NEWLY GENERATED LEADERS
no_government.0002 = {
    type = character_event
 	hidden = yes
 	immediate = {        
        clear_traits = yes
        add_trait = wastelands
    }
 }

 no_government.0003 = {    
    type = character_event
	hidden = yes
	
	trigger = {
		OR = {
			scope:host = {
				has_culture = culture:wastelands
			}
			scope:guest = {
				has_culture = culture:wastelands
			}
		}		
     }
	immediate = {
            scope:guest = {            
			    #trigger_event = no_government.0001
				death = {
					death_reason = death_vanished
				}	
			}
	}
}

no_government.0004 = {
    type = character_event
	hidden = yes
	
	root.province = {
		culture = culture:wastelands
	}

	immediate = {
		create_character = {
				template = servant_character
				dynasty = none
				location = root.province
				gender = male
				save_scope_as = peasant
			}
		scope:peasant = {
			remove_traits_wastelands_effect = yes
		}
		root.primary_heir = scope:peasant		
	}
}
