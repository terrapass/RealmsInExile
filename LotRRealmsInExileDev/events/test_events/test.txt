#Test events

namespace = test

test.1 = {
	type = character_event
	title = LOREM_IPSUM_TITLE
	desc = LOREM_IPSUM_DESCRIPTION
	theme = realm
	orphan = yes
	left_portrait = root

	immediate = {
		title:e_golden_realm_rhun = {
			if = {
				limit = { NOT = { has_variable = stability } }
				set_variable = { name = stability value = 50 }
			}
		}
	}

	option = {
		name = LOREM_IPSUM_TITLE
		title:e_golden_realm_rhun = {
			change_variable = { name = stability add = 10 }
		}
	}
	
	option = {
		name = LOREM_IPSUM_TITLE
		title:e_golden_realm_rhun = {
			change_variable = { name = stability subtract = 30 }
		}
	}
}

### LotR - Artifact Crafting ###
# This test event adds a list of variables to the character
# to test whether the modifier list is working properly.
#
# Might keep this here for whenever we implement court artifact
# crafting as-well.
# Jaco_Daan (the Snail Guy)

test.2 = {
	type = character_event
	title = LOREM_IPSUM_TITLE
	desc = LOREM_IPSUM_DESCRIPTION
	theme = realm
	orphan = yes
	left_portrait = root

	immediate = {
		set_variable = {
			name = used_material
			value = flag:silver_crafting
		}
		add_to_variable_list = {
			name = artifact_crafting_modifiers
			target = var:used_material
		}
		set_variable = {
			name = base_stats_modifiers
			value = flag:artifact_diplomacy_1
		}
		add_to_variable_list = {
			name = artifact_crafting_modifiers
			target = var:base_stats_modifiers
		}
		set_variable = {
			name = base_stats_modifiers
			value = flag:artifact_martial_1
		}
		add_to_variable_list = {
			name = artifact_crafting_modifiers
			target = var:base_stats_modifiers
		}
	}

	option = {
		name = LOREM_IPSUM_TITLE
	}
}
