saruman.0100 = {
	type = character_event
	title = religious_decision.0401.t
#	title = saruman.0100.t
	desc = {
		desc = religious_decision.0401.desc.beginning
#		desc = saruman.0100.desc.beginning
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:divination_outcome = flag:great_battle
				}
				desc = religious_decision.0401.desc.great_battle
#				desc = saruman.0100.desc.great_battle
			}
			triggered_desc = {
				trigger = {
					scope:divination_outcome = flag:hard_work
				}
				desc = religious_decision.0401.desc.hard_work
#				desc = saruman.0100.desc.hard_work
			}
			triggered_desc = {
				trigger = {
					scope:divination_outcome = flag:betrayal
				}
				desc = religious_decision.0401.desc.betrayal
#				desc = saruman.0100.desc.betrayal
			}
		}
	}
	theme = faith
	override_background = {
		event_background = corridor_night
	}
	left_portrait = root

	immediate = {
		remove_character_flag = divining_the_stars

		random_list = {
			20 = {
				save_scope_value_as = {
					name = divination_outcome
					value = flag:great_battle
				}
			}
			20 = {
				save_scope_value_as = {
					name = divination_outcome
					value = flag:hard_work
				}
			}
			20 = {
				save_scope_value_as = {
					name = divination_outcome
					value = flag:betrayal
				}
			}
		}
	}

	option = {
		name = religious_decision.0401.a
#		name = saruman.0100.a

		if = {
			limit = {
				scope:divination_outcome = flag:great_battle
			}
			add_character_modifier = {
				modifier = astrology_great_battle
				years = 10
			}
		}
		else_if = {
			limit = {
				scope:divination_outcome = flag:hard_work
			}
			add_character_modifier = {
				modifier = astrology_hard_work
				years = 10
			}
		}
		else_if = {
			limit = {
				scope:divination_outcome = flag:betrayal
			}
			add_character_modifier = {
				modifier = astrology_betrayal
				years = 10
			}
		}
	}
}

saruman.0501 = {
	hidden = yes

#	trigger = {
#		culture = culture:istari
#	}


	immediate = {
		if = {
			limit = {
				culture = culture:istari
				NOT = { has_trait = istari }
			}
			set_culture = culture:isengard_dunlending

			}

		}
	}
}
