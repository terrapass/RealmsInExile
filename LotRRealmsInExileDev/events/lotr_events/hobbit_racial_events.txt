namespace = hobbit_racial

hobbit_racial.0011 = { # Forced conversion to hobbit government type
	scope = none
	hidden = yes
	
	trigger = {
		is_hobbit = yes
		NOT = { government_has_flag = government_is_hobbit }
	}

	immediate = {
		change_government = hobbit_government
	}
}

hobbit_racial.2 = {
	hidden = yes

	immediate = {
		add_gold = ongoing_colonies_upkeep

		if = {
			limit = {
				has_character_modifier = wilderness_emporer_over_base_limit_negative
			}
			remove_character_modifier = wilderness_emporer_over_base_limit_negative
			add_character_modifier = {
				modifier = wilderness_count_over_base_limit_negative
			}
		}
		else_if = {
			limit = {
				has_character_modifier = wilderness_count_over_base_limit_negative
			}
			remove_character_modifier = wilderness_count_over_base_limit_negative
			add_character_modifier = {
				modifier = wilderness_duke_over_base_limit_negative
			}
		} 
		else_if = {
			limit = {
				has_character_modifier = wilderness_duke_over_base_limit_negative
			}
			remove_character_modifier = wilderness_duke_over_base_limit_negative
			add_character_modifier = {
				modifier = wilderness_king_over_base_limit_negative
			}
		}
		else_if = {
			limit = {
				has_character_modifier = wilderness_king_over_base_limit_negative
			}
			remove_character_modifier = wilderness_king_over_base_limit_negative
			add_character_modifier = {
				modifier = wilderness_emporer_over_base_limit_negative
			}
		} 
		else = {
			add_character_modifier = {
				modifier = wilderness_count_over_base_limit_negative
			}
		}
	}
}
