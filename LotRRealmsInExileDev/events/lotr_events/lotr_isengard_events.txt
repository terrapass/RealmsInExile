		character = root
		animation = personality_rational
	}
	lower_right_portrait = scope:beren

		character = scope:frealaf
		animation = brave
	}

	trigger = { is_independent_ruler = yes }

		trigger_event = { # Many Coloured.
			id = isengard.0010
			days = { 900 1800 }
		}
		# ToDo: Add Ring Maker event.
	}

	option = { # The rule of Isengard and its secrets are now mine.
		name = isengard.0001.a
		custom_tooltip = isengard.0001.a.tt
		character:lineofsaruman = { # Saruman receives Palantír of Orthanc.
			create_artifact_palantir_orthanc_effect = { OWNER = this }
		}
		ai_chance = { base = 100 }
	}
}
