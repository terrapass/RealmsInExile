realms_gondor_filler.100 = { # A haradrim vessel shipwrecked a few miles from the coast, and your men caught one survivor. His fate is up to you to decide.
	type = character_event
	title = realms_gondor_filler.100
	desc = realms_gondor_filler.100.desc
	theme = war
	override_background = {     
      event_background = docks
	}

	left_portrait = {
		character = root
		animation = personality_honorable
	}

	right_portrait = {
		character = cp:councillor_marshal
		animation = war_attacker
	}

	lower_center_portrait = {
		character = scope:survivor
		animation = war_attacker
	}

	trigger = { 
		NOT = {
			has_character_flag = washed_ashore
		}
		OR = {
			location = { 
				geographical_region = middleearth_west_gondor_centralfiefdoms 
			}
			location = { 
				geographical_region = middleearth_west_gondor_easternfiefdoms 
			}
		}
		root.primary_title = {
			is_coastal = yes
		}
		is_incapable = no
		is_imprisoned = no
	}

	immediate = {	#for redundance
		add_character_flag = washed_ashore
		create_character = {
			save_scope_as = survivor
			name = Drajze_e_n
			age = 20
			gender = male
			trait = humble
			trait = honest
			trait = calm
			health = 5
			fertility = 50
			employer = root
			faith = faith:
			random_culture_in_group = culture:lotr_haradrim
			dynasty = none
			diplomacy =  15
		}
		root = {
			imprison = { 
				target = scope:survivor # there should be a reason = line here. I have literally no fucking idea what goes with this, apparently its a modifier but I can't find any that would fit? perhaps Vierwood or someone else can fill this if it's even needed (btw love u Vier :3)
				type = house_arrest
			}
		}
	}

	option = {	# This is war. If the roles were reversed, he would not show us mercy.
		name = realms_gondor_filler.1001
		custom_tooltip = realms_gondor_filler.1001_tt
		trigger = {
			NOR = {
				has_trait = compassionate
				has_trait = forgiving
				has_trait = temperate
				has_trait = trusting
			}
		}
		if = {
			limit = {
				OR = {
					has_trait = sadistic
					has_trait = arbitrary
					has_trait = callous
				}
			}
			add_stress = -60
		}
		scope:survivor = {
			death = { 
				killer = root 
				death_reason = death_dungeon
			}
		}
	}

	option = {	# Relieve him of his chains, give him some water and bread.
		name = realms_gondor_filler.1002
		custom_tooltip = realms_gondor_filler.1002_tt
		if = {
			limit = {
				OR = {
					has_trait = compassionate
					has_trait = forgiving
					has_trait = temperate
					has_trait = trusting
				}
			}
			add_stress = -60
		}
		trigger_event = {
			id = realms_gondor_filler.101
			days = 5
		}
	}
}
