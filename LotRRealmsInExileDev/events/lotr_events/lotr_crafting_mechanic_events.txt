	title = crafting_mechanic.0001.t
	theme = learning
	desc = {}

		widget = { # Name
			is_shown = { has_variable = artifact_name }
			gui = "lotr_crafting_mechanic_name_widget"

			setup_scope = { save_scope_as = text_target }
			controller = {

				data = { key = custom_book_name default = authorship.display.name }

	}

	option = { # Set artifact description
        name = crafting_mechanic.0001.a
	}

	after = {
		remove_variable = artifact_name
	}
}
